kb

Keyboard input listener.
- EventEmitter: https://github.com/primus/eventemitter3
- Key code info: https://keycode.info/
Source:
  • src/utils/kb.js, line 53
Example
kb.onKeyUp(this, this.myListenerFn, 'ENTER', 'q', 32);
// ...
myListenerFn(key, keyCode, ev){
 if (keyCode == 13){
   // ENTER pressed
 } else if (key == 'SPACE'){
   // SPACE PRESSED
 }
}
// ...
kb.off(this);
